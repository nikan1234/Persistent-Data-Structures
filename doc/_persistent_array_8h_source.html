272 template <class... Args> [[nodiscard]] PersistentArray emplaceBack(Args &&...args) const {
273 auto &root = findOrCreateRoot();
274 CONTRACT_ASSERT(node_);
275
276 auto origin = node_;
277 if (root.contains(size_))
279 origin = PersistentNode::makeChangeSet(node_, size_, std::forward<Args>(args)...);
280 else
282 root.extend(std::forward<Args>(args)...);
283
284 return modify(std::move(origin), size_ + 1);
285 }

315 [[nodiscard]] PersistentNode &findOrCreateRoot() const {
316 if (!node_)
317 return SAFE_DEREF(node_ = PersistentNode::makeRoot());
318
319 auto foundNode = node_;
320 while (foundNode && !foundNode->isRoot())

326 [[nodiscard]] PersistentArray modify(typename PersistentNode::Ptr newNode,
327 const std::size_t newSize) const {
328
329 const auto undo = [size = size_, node = node_](auto manager) {
330 return PersistentArray{size, node, std::move(manager)};
331 };
332 const auto redo = [size = newSize, node = std::move(newNode)](auto manager) {
333 return PersistentArray{size, node, std::move(manager)};
334 };
335 return redo(this->undoManager().pushUndo(Undo::createAction<PersistentArray>(undo, redo)));
336 }

339 void reRootModificationTree() const {
340 std::stack<typename PersistentNode::Ptr> path;
341
342 auto root = node_;
343 while (!root->isRoot()) {
344 path.push(root);
345 root = root->reparent(nullptr);
346 }
347
349 while (!path.empty()) {
350 root->siftUpRoot(path.top());
351 root->reparent(std::move(path.top()));
352 path.pop();
353 root = root->parent();
354 }
355 }
