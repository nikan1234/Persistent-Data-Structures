16struct PersistenceListOrder {
18 const double weight_border = 2000000000000;
20 std::list<int> list;
22 std::vector<std::list<int>::iterator> handles;
24 std::vector<double> weight_true;
26 std::vector<double> weight_reverse;
27 PersistenceListOrder(){};
28
32 int add(
33 int parent) {
34 if (list.size() == 0) {
35 handles.push_back(list.insert(list.end(), 1));
36 handles.push_back(handles[0]);
37 list.insert(list.end(), -1);
38 weight_true.push_back(-weight_border);
39 weight_true.push_back(-weight_border);
40 weight_reverse.push_back(weight_border);
41 weight_reverse.push_back(weight_border);
42 return 1;
43 }
44 CONTRACT_EXPECT(parent <= handles.size());
45 auto next_parent_handle = handles[parent];
46 ++next_parent_handle;
47 double parent_value = weight_true[parent];
48 double next_parent_value;
49 int next_parent = *next_parent_handle;
50 if (next_parent > 0) {
51 next_parent_value = weight_true[next_parent];
52 } else {
53 next_parent_value = weight_reverse[-next_parent];
54 }
55 int new_version = handles.size();
56 auto version_handle = list.insert(next_parent_handle, new_version);
57 handles.push_back(version_handle);
58 ++version_handle;
59 list.insert(version_handle, -new_version);
60 double true_weight = parent_value + (next_parent_value - parent_value) / 3;
61 double true_reverse = parent_value + 2 * (next_parent_value - parent_value) / 3;
62 weight_true.push_back(true_weight);
63 weight_reverse.push_back(true_reverse);
64 if (true_weight == true_reverse) {
65 double step = weight_border / weight_true.size();
66 double cur = -weight_border;
67 for (auto i : list) {
68 if (i < 0) {
69 weight_reverse[-i] = cur;
70 } else {
71 weight_true[i] = cur;
72 }
73 cur += step;
74 }
75 }
76 return new_version;
77 }
78
83 bool less(int l, int r) {
84 CONTRACT_EXPECT(abs(l) < weight_true.size());
85 CONTRACT_EXPECT(abs(r) < weight_true.size());
86 double weight_l, weight_r;
87 if (l < 0) {
88 weight_l = weight_reverse[-l];
89 } else {
90 weight_l = weight_true[l];
91 }
92 if (r < 0) {
93 weight_r = weight_reverse[-r];
94 } else {
95 weight_r = weight_true[r];
96 }
97 return weight_l < weight_r;
98 }

423 cur_last->getLast(version), cur_next, PersistenceCmpByListVersion(listOrder_));
424 cur_new_node->copyNextAfter(cur_last, version);
